            <body>
                <div class="container">
                    <h1>Salarios Promedio por Departamento</h1>
                    
                    <div class="stats-grid">
                        <xsl:for-each select="empresa/departamentos/departamento">
                            <xsl:sort select="nombre"/>
                            <xsl:variable name="departamento-id" select="@id"/>
                            <xsl:variable name="departamento-nombre" select="nombre"/>
                            <xsl:variable name="empleados-departamento" select="//empleado[id_departamento = $departamento-id]"/>
                            
                            <xsl:if test="count($empleados-departamento) > 0">
                                <xsl:variable name="total-salarios" select="sum($empleados-departamento/salario)"/>
                                <xsl:variable name="cantidad-empleados" select="count($empleados-departamento)"/>

                                    <div class="stat-header">
                                        <xsl:value-of select="$departamento-nombre"/>
                                    </div>
                                    <div class="stat-content">
                                        <div class="stat-value">
                                            $<xsl:value-of select="format-number($salario-promedio, '#,##0.00')"/>
                                        </div>
                                        <div class="stat-label">Salario Promedio</div>
                                        <div class="stat-details">
                                            <xsl:value-of select="$cantidad-empleados"/> empleado(s)
                                        </div>
                                    </div>
                                </div>
                            </xsl:if>
